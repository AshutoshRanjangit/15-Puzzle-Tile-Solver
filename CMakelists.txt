cmake_minimum_required(VERSION 3.16)
project(Puzzle15Solver CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release)
endif()

set(CMAKE_CXX_FLAGS_DEBUG   "-O0 -g -Wall -Wextra")
set(CMAKE_CXX_FLAGS_RELEASE "-O3 -DNDEBUG -Wall -Wextra")

message(STATUS "Build type: ${CMAKE_BUILD_TYPE}")

add_library(puzzle_core STATIC
    src/puzzle_state.cpp
    src/solvability.cpp
    src/heuristic.cpp
    src/bfs_solver.cpp
    src/astar_solver.cpp
)
target_include_directories(puzzle_core PUBLIC include)

add_executable(puzzle15 src/main.cpp)
target_link_libraries(puzzle15 PRIVATE puzzle_core)

add_executable(benchmark benchmarks/benchmark.cpp)
target_link_libraries(benchmark PRIVATE puzzle_core)

enable_testing()

set(TEST_NAMES
    test_puzzle_state
    test_solvability
    test_heuristic
    test_bfs_solver
    test_astar_solver
)

foreach(test_name ${TEST_NAMES})
    add_executable(${test_name} tests/${test_name}.cpp)
    target_link_libraries(${test_name} PRIVATE puzzle_core)
    target_compile_options(${test_name} PRIVATE -UNDEBUG)
    add_test(NAME ${test_name} COMMAND ${test_name})
endforeach()
